## 构造器实现

注意点：

1. 考虑 `executor` 执行时机
2. 考虑多次执行 `resolve()`、`reject()` 状态变更问题
3. 考虑 `executor` 执行异常问题，包括同步异常、异步异常的场景
4. 考虑 `resolve(promise)` 的问题

用例：

```js
class MyPromise {
  // TODO: ...
}

// 2️⃣
const p1 = new Promise(resolve => {
  resolve(1)
})

// 3️⃣
const p2 = new Promise((resolve, reject) => {
  resolve(1)
  reject(2)
})

// 4️⃣
const p3 = new Promise(resolve => {
  resolve(1)
  throw 2
})

// 5️⃣
const p4 = new Promise(resolve => {
  setTimeout(() => {
    resolve(1)
  })
})

// 6️⃣
const p4 = new Promise(resolve => {
  setTimeout(() => {
    resolve(1)
  })
  throw 2
})
```

实现：

```js
const STATE = {
  PENDING: 'pending',
  FULFILLED: 'fulfilled',
  REJECTED: 'rejected',
}

class MyPromise {
  #state = STATE.PENDING
  #result = undefined

  constructor(executor) {
    const resolve = value => {
      this.#stateChange(STATE.FULFILLED, value)
    }

    const reject = reason => {
      this.#stateChange(STATE.REJECTED, reason)
    }

    try {
      executor(resolve, reject)
    } catch (err) {
      // 处理异常，如 throw 'xxx'
      reject(err)
    }
  }

  #stateChange(state, result) {
    // 处理多次执行 resolve、reject 的请求
    if (this.#state !== STATE.PENDING) return
    
    this.#state = state
    this.#result = result
  }

  // TODO: 下一步
  then() {}
}
```

## then 实现

注意点：

1. 考虑 `then()` 参数个数问题
2. 考虑 `then()` 及其回调的执行时机问题
3. 考虑 `then()` 参数不是函数的问题
4. 考虑 `then()` 返回类型问题
5. 考虑链式调用值穿透问题，比如 `promise.then().then().then(res => res)`
6. 考虑非链式多次调用问题，比如 `promise.then(res => res + 1); promise.then(res => res + 1)`
7. 当 Promise 实例处于不同状态（ `pending`、`fulfilled`、`rejected`），`then()` 执行时机问题

用例：

```js
const promise = new MyPromise(resolve => resolve(1))


// 1️⃣ 注意：调用 then 时，promise 可能已经是 fulfilled 或 rejected，也可能仍处于 pending
promise.then(res => res)

// 2️⃣
promise.then(res => res, err => err)

// 3️⃣
promise.then().then(res => res)

// 4️⃣
promise.then().then().then(res => res)

// 5️⃣ 注意，区别于 promise.then(res => res + 1).then(res => res + 1) 链式调用
promise.then(res => res + 1)
promise.then(res => res + 1)
```

实现：

```js
class MyPromise {
  #state = STATE.PENDING
  #result = undefined
  #thenables = []

  constructor(executor) {
    const resolve = value => {
      this.#stateChange(STATE.FULFILLED, value)
    }

    const reject = reason => {
      this.#stateChange(STATE.REJECTED, reason)
    }

    try {
      executor(resolve, reject)
    } catch (err) {
      reject(err)
    }
  }
  
  then(onfulfilled, onrejected) {
    // 返回新的 Promise
    return new MyPromise((resolve, reject) => {
      // 由于一个 promise 可以多次调用 then 方法，这里需要一个队列维护这些回调方法
      // 比如 `promise.then(); promise.then()`（非链式调用）
      this.#thenables.push({ onfulfilled, onrejected, resolve, reject })

      // 如果 Promise 状态已变更，执行调用 then 则立即执行（指立刻放入微队列），否则等 #stateChange 处触发
      // 注意，如果是 then 回调应该在微任务队列里执行，而不是同步执行。
      this.#run()
    })
  }

  #stateChange(state, result) {
    if (this.#state !== STATE.PENDING) return
    
    this.#state = state
    this.#result = result

    // 对于 setTimout(() => {resolve(1)}) 这类情况，then 回调应该在其 Promise 状态变化时执行
    this.#run()
  }
  
  #run() {
    // 由于 then 回调会马上调用一次 run 方法，这里要做状态判断
    if (this.#state === STATE.PENDING) return

    // 把队列里的东西，按先进先出的顺序，拿出来执行
    while (this.#thenables.length) {
      const { onfulfilled, onrejected, resolve, reject } = this.#thenables.shift()

      const callback = this.#state === STATE.FULFILLED ? onfulfilled : onrejected
      this.#handleThenableCallback(callback, resolve, reject)
    }
  }

  #handleThenableCallback(callback, resolve, reject) {
    // 处理值穿透，比如 promise.then().then(res => res)
    if (typeof callback !== 'function') {
      // 状态变更要放到微队列里面
      queueMicrotask(() => {
        if (this.#state === STATE.FULFILLED) {
          resolve(this.#result)
        } else {
          reject(this.#result)
        }
      })

      return
    }

    // 考虑 reject 可能是一个 undefined 的问题，比如 promise.then(res => {}) 省略第二个参数
    queueMicrotask(() => {
      try {
        // 将 then 回调执行结果传到下一个 then 里面
        const result = callback(this.#result)
        resolve(result)
      } catch (err) {
        reject(err)
      }
    })
  }
}
```

## 完整示例

```js
const STATE = {
  PENDING: 'pending',
  FULFILLED: 'fulfilled',
  REJECTED: 'rejected',
}

class MyPromise {
  #state = STATE.PENDING
  #result = undefined
  #thenables = []

  constructor(executor) {
    const resolve = value => {
      this.#stateChange(STATE.FULFILLED, value)
    }

    const reject = reason => {
      this.#stateChange(STATE.REJECTED, reason)
    }

    try {
      executor(resolve, reject)
    } catch (err) {
      reject(err)
    }
  }

  then(onfulfilled, onrejected) {
    return new MyPromise((resolve, reject) => {
      this.#thenables.push({ onfulfilled, onrejected, resolve, reject })
      this.#run()
    })
  }

  #stateChange(state, result) {
    if (this.#state !== STATE.PENDING) return

    this.#state = state
    this.#result = result

    this.#run()
  }

  #run() {
    if (this.#state === STATE.PENDING) return

    while (this.#thenables.length) {
      const { onfulfilled, onrejected, resolve, reject } = this.#thenables.shift()
      
      const callback = this.#state === STATE.FULFILLED ? onfulfilled : onrejected
      this.#handleThenableCallback(callback, resolve, reject)
    }
  }

  #handleThenableCallback(callback, resolve, reject) {
    if (typeof callback !== 'function') {
      queueMicrotask(() => {
        if (this.#state === STATE.FULFILLED) {
          resolve(this.#result)
        } else {
          reject(this.#result)
        }
      })

      return
    }

    queueMicrotask(() => {
      try {
        const result = callback(this.#result)
        resolve(result)
      } catch (err) {
        reject(err)
      }
    })
  }
}
```
